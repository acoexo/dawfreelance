import java.util.Scanner;

public class CalcEntry{
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        System.out.println("Este programa sirve para gestionar las entradas del campo de fútbol de Gran Canaria");
        /*Ignorar el while ya que es solo una implementación para la automatización para no tener que darle 
        constantemente a la flechita de run*/
        int repetir = 1;
        while (repetir == 1) {
            System.out.println("___________________________________");
            System.out.println("_______________________"+"\n"
                    +"|1 Naciente:         12€|"+"\n"
                    +"|2 Curva:            18€|"+"\n"
                    +"|3 Sur:              20€|"+"\n"
                    +"|4 Tribuna:          25€|"+"\n"
                    +"|5 Tribuna especial: 35€|"+"\n"
                    +"|6 VIP:              60€|"+"\n"
                    +"_______________________");
            System.out.println("                       ");
            System.out.println("(Por favor seleccione un número del 1 al 6 correspondiente a la posición en orden descendente)");
            System.out.print("¿En qué lugar se va a posicionar?: ");
            int lugar = in.nextInt();
            if (lugar>6||lugar<1){
                System.out.println("Esta zona no existe, por favor, introduce una zona válida");
            }else {
                System.out.print("¿Cuantas entradas quiere?: ");
                int quantity = in.nextInt();
                int total;
                switch (lugar) {
                    case 1:
                        total = (12 * quantity);
                        System.out.println("Ha elegido la zona Naciente, precio: " + total + " €");
                        break;
                    case 2:
                        total = (18 * quantity);
                        System.out.println("Ha elegido la zona Curva, precio: " + total + " €");
                        break;
                    case 3:
                        total = (20 * quantity);
                        System.out.println("Ha elegido la zona Sur, precio: " + total + " €");
                        break;
                    case 4:
                        total = (25 * quantity);
                        System.out.println("Ha elegido la zona Tribuna, precio: " + total + " €");
                        break;
                    case 5:
                        total = (35 * quantity);
                        System.out.println("Ha elegido la zona Tribuna especial, precio: " + total + " €");
                        break;
                    case 6:
                        total = (60 * quantity);
                        System.out.println("Ha elegido la zona VIP, precio: " + total + " €");
                        break;
                }
            }

            //Ignorar esto también
            System.out.print("¿Quieres repetir? introduce 1 si quieres o un 0 si no: ");
            repetir = in.nextInt();
        }
        //Y esto también
        if (repetir==0) {
            System.out.print("Adiós");
        }
    }
}
